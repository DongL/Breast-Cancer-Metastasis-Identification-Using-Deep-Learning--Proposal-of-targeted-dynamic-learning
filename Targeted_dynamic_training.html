def read_data_from_cloud(bucket_name, file_name, x_or_y):

    """
    Args:
        bucket_name: name of google cloud storage bucket;
        file_name: name of the file name;
        x_or_y: 'x' for features and 'y' for target;
    Return:
        feature data or target data.
    """

    client = storage.Client()
    bucket = client.get_bucket(bucket_name)
    blob = bucket.get_blob(file_name)
    content = blob.download_as_string()
    
    if x_or_y == 'x':
        data = HDF5Matrix(BytesIO(content), 'x')  # return a HDF5matrix stream
    else:
        data = h5py.File(BytesIO(content), 'r')
        data = data.get('y')
        data = np.array(data).flatten()  # return a 1-d array for binary targets
        
    return data

# Callback
# custom callback for logging train loss, train auc, train accuracy, valid loss, valid auc and valid accuracy.
from tensorflow.keras.callbacks import ModelCheckpoint, EarlyStopping

class LossHistory(tf.keras.callbacks.Callback):
    def on_train_begin(self, logs={}):
        self.losses = []
        self.accuracy = []
        self.auc_roc = []
        self.val_loss = []
        self.val_accuracy = []
        self.val_auc_roc = []

    def on_batch_end(self, batch, logs={}):
        self.losses.append(logs.get('loss'))
        self.accuracy.append(logs.get('accuracy'))
        self.auc_roc.append(logs.get('auc_roc'))
        self.val_loss.append(logs.get('val_loss'))
        self.val_accuracy.append(logs.get('val_accuracy'))
        self.val_auc_roc.append(logs.get('val_auc_roc'))

        
# create a callback to save the model weights after each epoch if the validation loss decreased
file_path = './resnet50'
history = LossHistory()
earlyStopping = tf.keras.callbacks.EarlyStopping(monitor='val_loss', patience=5)
callback = [ModelCheckpoint(filepath=file_path, monitor='val_loss', verbose=1, save_best_only=True),
           history, earlyStopping]

# separate samples into pos or neg groups
def sampling_from_neg_pos_grps(model, x_test, y_test, weight = 0.01, equal_size = True):

    # Sampling
    np.random.seed(96)
    size = len(y_test) * weight
    index_random = np.random.choice(range(len(y_test)), int(size), replace = False)
    x_test_ = x_test[:]
    
    x_test = x_test[:]/255.
    x_test = x_test[index_random]
    y_test = y_test[index_random]
    
    # Positive and negative indice 
    pred = model.predict(x_test)
    binary_pred = np.where(pred.flatten() > 0.5, 1, 0)
    index_pos = np.where(binary_pred == y_test)[0]
    index_neg = [i for i in range(len(y_test)) if i not in index_pos]
    
    # Positive and negative groups
    group_pos = x_test_[index_random][index_pos].astype(np.float32)
    group_neg = x_test_[index_random][index_neg].astype(np.float32)
    
    # Accuracy
    print(len(index_pos)/len(y_test))
    
    # 
    if equal_size:
        ind = np.random.choice(len(index_pos), len(index_neg), replace = False)
        index_pos = index_pos[ind]
        group_pos = group_pos[ind]
    
    return group_pos, group_neg, index_pos, index_neg, index_random

class Encoder(tf.keras.layers.Layer):
    def __init__(self,
               latent_dim=10,
               intermediate_dim=[256, 128],
               name='encoder',
               **kwargs):
        super(Encoder, self).__init__(name=name, **kwargs)
        self.LayerFlatten = tf.keras.layers.Flatten()
        self.LayerDense1 = tf.keras.layers.Dense(intermediate_dim[0], activation='relu')
        self.LayerDense2 = tf.keras.layers.Dense(intermediate_dim[1], activation='relu')
        self.LayerDense3 = tf.keras.layers.Dense(latent_dim)
    
    
    def call(self, inputs):
#         x = self.LayerFlatten(inputs)
        x = self.LayerDense1(inputs)
        x = self.LayerDense2(x)
        x = self.LayerDense3(x)
        return x

    def __init__(self,

        super(Decoder, self).__init__(name=name, **kwargs)

        self.LayerDense2 = tf.keras.layers.Dense(intermediate_dim[0], activation='relu')

        x = self.LayerDense1(inputs)
        x = self.LayerDense2(x)
        x = self.LayerDense3(x)
        return x

# Calculate the kl divergence for valid dataset
from scipy.special import rel_entr, kl_div
from scipy.spatial.distance import cosine

def kl_divergence(p, q):
    return sum(p[i] * np.log2(p[i]/q[i]) for i in range(len(p)))

def get_kl_divergence(valid_latent, neg_latent):
    p_valid_latent = tf.nn.softmax(valid_latent)
    q_valid_latent = tf.nn.softmax(neg_latent)

    dist = np.sum(rel_entr(p_valid_latent, q_valid_latent), axis = 1)
    dist = np.mean(dist)
    return dist 


def get_cosine_dist(valid_latent, neg_latent):
    cosine_dist = [cosine(valid_latent, neg) for neg in neg_latent]
    return np.min(cosine_dist)


def get_probability(valid_latent_, neg_latent_):
    valid_latent100 = valid_latent_
    dist_neg = [get_cosine_dist(val, neg_latent_) for val in valid_latent100]
    dist_pos = [get_cosine_dist(val, pos_latent_) for val in valid_latent100]
    prob_neg = [1/neg / (1/neg + 1/pos) for neg, pos in zip(dist_neg, dist_pos)]
    return prob_neg

class SiameseBlock(tf.keras.layers.Layer):
    def __init__(self, units = [1024, 512], dropout_rate = 0.2, **kwargs):
        super().__init__(**kwargs)
        self.LayerDense3 = tf.keras.layers.Dense(units[1])
        self.LayerReshape = tf.keras.layers.Reshape([96, 96, 1])
        self.LayerFlatten = tf.keras.layers.Flatten()
        self.LayerConv1 = tf.keras.layers.Conv2D(filters = 3, kernel_size=3, padding= 'same', activation = 'relu')
        self.LayerConv2 = tf.keras.layers.Conv2D(filters = 64, kernel_size=3, padding= 'same', activation = 'relu')
    
    
    def call(self, x):
        x = self.LayerConv1(x)
        x = self.LayerFlatten(x)
        x = self.LayerDense3(x)
        return x
    


class Siamese(tf.keras.Model):
    '''
    Return:
        shape = (batch_size, probability_dim)
    '''
    def __init__(self):
        super().__init__()
        self.LayerSiamese = SiameseBlock()
        self.LayerDense = Dense(1, activation='sigmoid')
        
    def get_cosine_dist(self, x1_latent, x2_latent):
        cosine_dist = [cosine(l1, l2) for l1, l2 in zip(x1_latent, x2_latent)]
        return cosine_dist   

    def call(self, inputs):
        x1 = self.LayerSiamese(inputs[0])
        x2 = self.LayerSiamese(inputs[1])
        cosine_dist = self.get_cosine_dist(x1, x2)
        L1 = tf.abs((x1 - x2))
        z = self.LayerDense(L1)
        return z, cosine_dist

def train_model(model, epoch, loss_fn, dataset, optimizer_fn,  manager, learning_rate = 0.001, print_every = 1,  **kwargs):

    optimizer = optimizer_fn(learning_rate = learning_rate)

    # Create loss and evaluation metrics 
    loss_val = tf.keras.metrics.Mean(name = 'loss_val')
    accuracy_val = tf.keras.metrics.BinaryAccuracy(name='accuracy_val')

    loss_train = tf.keras.metrics.Mean(name = 'loss_train')
    accuracy_train = tf.keras.metrics.BinaryAccuracy(name='accuracy_train')
#     cosine_dist_train = tf.keras.metrics.Mean(name = 'cosine_dist')


    # Initialize metrics
    loss_train.reset_states()
    accuracy_train.reset_states()
#     cosine_dist_train.reset_states()
    
#     size = int(len(dataset[0])/2)
    size = len(dataset[0])

    for k in range(epoch):
#         for _, (x, y) in enumerate(train_dataset):
        for (x, y) in zip(dataset[0][:size], dataset[1][:size]):
            with tf.GradientTape() as type:
                # Model prediction
                yhat, cosine_dist = model(x)
#                 print(yhat)
#               # Define loss function
                def loss_fun(y, yhat):
                    epsilon = 1e-6
                    yhat = tf.clip_by_value(yhat, epsilon, 1 - epsilon)
                    batch_loss = -y * tf.math.log(yhat) - (1 - y) * tf.math.log(1 - yhat)
                    return batch_loss

                # Computer loss
                loss = loss_fn(y, yhat)
                loss_mean = tf.reduce_mean(loss)
                # loss = loss_fn(y, yhat) 

                # Record metrics
                loss_train.update_state(loss_mean)
                accuracy_train.update_state(y, yhat)
#                 cosine_dist_train.update_state(tf.reduce_mean(cosine_dist))

            # Calculate gradients of weights and biases
            grad  = type.gradient(loss_mean, model.trainable_weights)
            
            # Apply gradients
            optimizer.apply_gradients(zip(grad, model.trainable_weights))

        # Check point
        if manager != None:
            ckpt.step.assign_add(1)
            manager.save()

        if k % print_every == 0 or k == epoch - 1:        
            print('Epoch', k + 1)
            if not kwargs.keys():
                print(f"Train loss:{loss_train.result(): {0}.{4}f}", 
                      f"Train Acc:{(accuracy_train.result() * 100): {0}.{2}f}", 
#                       f"Cosine Distance:{cosine_dist_train.result(): {0}.{4}f}"
                      )
            else:
                # Initialize metrics
                loss_val.reset_states()
                accuracy_val.reset_states()

                # Calculate validation metrics
                for (x, y) in zip(dataset[0][size:], dataset[1][size:]):
                    yhat = model(x)
                    loss = loss_fn(y, yhat)
                    loss_mean = tf.reduce_mean(loss)

                    loss_val.update_state(loss_mean)
                    accuracy_val.update_state(y, yhat)

                print(
                      f"Train loss:{loss_train.result(): {0}.{4}f}", 
                      f"Train Acc:{(accuracy_train.result() * 100): {0}.{2}f}", 
                      # f"Train SNR:{batch_SNR_train: {0}.{2}f}", 
                      f"Val Loss:{loss_val.result(): {0}.{4}f}", 
                      f"Val Acc:{(accuracy_val.result() * 100): {0}.{2}f}",
                      # f"Valid SNR:{batch_SNR_val: {0}.{2}f}", 
                      )

    return model, loss_train, loss_val

def get_per_sample_similarity_and_probability(model, valid, y_valid, dataset_neg_, dataset_pos_):
    similarity_scores_neg = dict()
    similarity_scores_pos = dict()
    
    # Similarity scores for dataset_neg_
    for i, v in enumerate(valid):
        scores = list()
        for neg in dataset_neg_:
            pair = np.array([v[np.newaxis, :], neg[np.newaxis, :]])
            scores.append(model(pair)[0].numpy().flatten()[0])    
        similarity_scores_neg[i] = scores
    
    # Similarity scores for dataset_pos_
    for i, v in enumerate(valid):
        scores = list()
        for neg in dataset_pos_:
            pair = np.array([v[np.newaxis, :], neg[np.newaxis, :]])
            scores.append(model(pair)[0].numpy().flatten()[0])    
        similarity_scores_pos[i] = scores
         
    # Probability of being in negative group
    prob = list()
    for x, y in zip(similarity_scores_neg, similarity_scores_pos):
        score_neg = np.mean(similarity_scores_neg[x]) 
        score_pos = np.mean(similarity_scores_pos[y])
        delta = np.abs(score_neg - score_pos)
        prob.append((score_neg / (score_neg + score_pos)))
        
    # Prediction Performance ()
    prob = np.array(prob)
    index = np.where(prob > 0.6)
    print(len(index))

    return similarity_scores_neg, similarity_scores_pos, prob, index
