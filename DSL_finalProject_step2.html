# function to shuffle the data to create minibatch 
def shuffle_batch(X, y, batch_size):
    rnd_idx = np.random.permutation(len(X))
    n_batches = len(X) // batch_size
    # import pdb; pdb.set_trace()
    for batch_idx in np.array_split(rnd_idx, n_batches):
        X_batch, y_batch = np.array(X)[batch_idx], y[batch_idx]
        yield X_batch, y_batch

"""

utility function to create static tensorflow graph for hyperparameter tuning

"""

 

def create_mlp_graph(n_inputs, n_outputs, n_hidden_layers=3, n_neurons_per_layer=1000, activation_function=tf.nn.relu, initiation_method='xavier_normal',

                 learning_rate=0.001, optimize_method='adam', dropout_rate=0.5):

  """

  Args:

    n_inputs: no of features of raw input data;

    n_outputs: no of outputs of last layer;

    n_hidden_layers: no of hidden layers;

    n_neurons_per_layer: no of neurons per layer;

    activation_function: activation function of hidden layers;

    initiation_method: initialization method of weights of hidden layers;

    learning_rate: learning rate;

    optimize_method: optimization method for gradient descent;

    dropout_rate: dropout rate for hidden layers.

  Return:

    g: a tennsorflow graph;

    loss: operation for loss function;

    accuracy: accuracy;

    training_op: operation for training the model using some optimization method;

    X: placeholder for raw input data;

    y: placeholder for output data;

    y_proba: predicted probability.

  """

  def build_fcc_layers(inputs, n_hidden, var_scope_name, init_method):

     

      with tf.variable_scope(var_scope_name):

       

          hidden_layers = {} # dictionary of hidden layers

          batch_normal_layers = {} # dictionary of batch normalize layers

          batch_act = {} # dictionary of activation layers

          hidden_dropout_layers = {} # dictionary of dropout layers

 

          for i in range(n_hidden_layers):

            if i == 0:

              hidden_layers['1'] = tf.layers.dense(inputs, n_neurons_per_layer, kernel_initializer=init_method, name="hidden_1")

              batch_normal_layers['1'] = tf.layers.batch_normalization(hidden_layers['1'], training=training, momentum=0.9)

              batch_act['1'] = activation_function(batch_normal_layers['1'])

              hidden_dropout_layers['1'] = tf.layers.dropout(batch_act['1'], dropout_rate, training=training)

            else:

              hidden_layers[str(i+1)] = tf.layers.dense(hidden_dropout_layers[str(i)], n_neurons_per_layer,

                                                               kernel_initializer=init_method, name="hidden_{}".format(i+1))

              batch_normal_layers[str(i+1)] = tf.layers.batch_normalization(hidden_layers[str(i+1)], training=training, momentum=0.9)

              batch_act[str(i+1)] = activation_function(batch_normal_layers[str(i+1)])

              hidden_dropout_layers[str(i+1)] = tf.layers.dropout(batch_act[str(i+1)], dropout_rate, training=training)

   

      return hidden_dropout_layers[str(n_hidden_layers)]

 

  g = tf.Graph()

 

  with g.as_default():

    X = tf.placeholder(tf.float32, shape=(None, n_inputs), name="X")

    y = tf.placeholder(tf.float32, shape=(None, n_outputs), name="y")

    training = tf.placeholder_with_default(False, shape=(), name='training')

    alpha = tf.Variable([1., 1., 1., 1., 1.])

   

    if initiation_method == 'xavier_normal':
      init_method = tf.contrib.layers.xavier_initializer(uniform=False)
    elif initiation_method == 'xavier_uniform':
      init_method = tf.contrib.layers.xavier_initializer(uniform=True)
    elif initiation_method == 'he_normal':
      init_method = tf.initializers.he_normal()
    elif initiation_method == 'he_uniform':
      init_method = tf.initializers.he_uniform()
    elif initiation_method == 'random_normal':
      init_method = tf.random_normal_initializer()
    elif initiation_method == 'random_uniform':
      init_method == tf.random_uniform_initializer()
    else:
      init_method = tf.initializers.TruncatedNormal()

    with tf.name_scope("dnn"):

      hidden_output_1 = build_fcc_layers(inputs=X, n_hidden=n_hidden_layers, var_scope_name='hidden_block_1', init_method=init_method)

      inputs_2 = tf.where(X > 1 - 1e-2, X, tf.zeros_like(X))

      hidden_output_2 = build_fcc_layers(inputs=inputs_2, n_hidden=n_hidden_layers, var_scope_name='hidden_block_2', init_method=init_method)

   

      inputs_3 = tf.where(X > 1 - 1e-3, X, tf.zeros_like(X))

      hidden_output_3 = build_fcc_layers(inputs=inputs_3, n_hidden=n_hidden_layers, var_scope_name='hidden_block_3', init_method=init_method)

   

      inputs_4 = tf.where(X > 1 - 1e-4, X, tf.zeros_like(X))

      hidden_output_4 = build_fcc_layers(inputs=inputs_4, n_hidden=n_hidden_layers, var_scope_name='hidden_block_4', init_method=init_method)

   

      inputs_5 = tf.where(X > 1 - 1e-5, X, tf.zeros_like(X))

      hidden_output_5 = build_fcc_layers(inputs=inputs_5, n_hidden=n_hidden_layers, var_scope_name='hidden_block_5', init_method=init_method)

       

        

#       hidden_outputs_stack = tf.stack([hidden_output_1, hidden_output_2, hidden_output_3,

#                                        hidden_output_4, hidden_output_5])   

#       hidden_output = tf.reshape(tf.matmul(tf.reshape(alpha, (1, 5)), tf.reshape(hidden_outputs_stack, (5, -1))),

#                                  (-1, 1000))

     

      hidden_output = alpha[0]*hidden_output_1 + alpha[1]*hidden_output_2 + alpha[2]*hidden_output_3 + \
                        alpha[3]*hidden_output_4 + alpha[4]*hidden_output_5

     

      logits = tf.layers.dense(hidden_output, n_outputs, activation=tf.nn.relu, kernel_initializer=init_method,

                             name="last")

 

    with tf.name_scope("loss"):

      xentropy = tf.nn.sigmoid_cross_entropy_with_logits(labels=y,logits=logits)

      loss = tf.reduce_mean(xentropy)

       

      y_proba = tf.nn.sigmoid(logits)  # prediction

      correct = tf.equal(tf.round(y_proba), y)

      accuracy = tf.reduce_mean(tf.cast(correct, tf.float32))

 

    with tf.name_scope("train"):

      if optimize_method == 'nesterov': # nesterov optimizer

        optimizer = tf.train.MomentumOptimizer(learning_rate=learning_rate, momentum=0.9, use_nesterov=True)

      elif optimize_method == 'adam': # adam optimizer

        optimizer = tf.train.AdamOptimizer(learning_rate=learning_rate)

      else: # momentum optimizer

        optimizer = tf.train.MomentumOptimizer(learning_rate=learning_rate, momentum=0.9)

      training_op = optimizer.minimize(loss)

 

  return g, loss, accuracy, training_op, X, y, y_proba, logits

"""

utility function to create static tensorflow graph for hyperparameter tuning

"""

 

def create_cnn_graph(n_inputs,

                 height,

                 width,

                 channels,

                 n_outputs,

                 conv_layers_hp=[{'kernel':32, 'ksize':[1, 3], 'stride':[1, 1], 'padding':'SAME'},

                                 {'kernel':64, 'ksize':[1, 3], 'stride':[1, 1], 'padding':'SAME'}],

                 pool_layers_hp=[{'ksize':[1, 1], 'stride':[1, 2], 'padding':'VALID'},

                                 {'ksize':[1, 1], 'stride':[1, 1], 'padding':'VALID'}],

                 pool_dropout_rate = 0.5,

                 fc_layers_hp=1024,

                 fc_dropout_rate = 0.5,

                 learning_rate=0.001,

                 average_pool=False):

  """

  Args:

    n_inputs: no of features of raw input data;

    height, width, channels：dimension of a single instance of X;

    n_outputs: no of outputs of last layer;

    conv_layers_hp: list of dictionary of hyperparameters of convalutional layers.

                    length of conv_layers has to be the same as pool_layer_hp;

    pool_layers_hp: list of dictionary of hyperparameters of max_pool layers.

                    length of pool_layers has to be the same as conv_layer_hp;

    pool_dropout_rate: dropout rate after the final pool layer;

    fc_layers_hp: no of neurons of every fully connected layers;

    fc_dropout_rate: dropout rate after the fully connected layer;

    learning_rate: learning rate;

    average_pool: True if average_pooling used before the last layer else False.

  Return:

    g: a tennsorflow graph

    loss: operation for loss function

    training_op: operation for training the model using some optimization method

    X: placeholder for raw input data;

    y: placeholder for output data.

  """

 

  def build_cnn_layers(inputs, n_blocks, var_scope_name):

     

      with tf.variable_scope(var_scope_name):

          conv_layers = {} # dictionary of conv layers

          pool_layers = {} # dictionary of max_pool layers

 

          for i in range(n_blocks):

            if i == 0:

              conv_layers['1'] = tf.layers.conv2d(inputs,

                                              filters=conv_layers_hp[0]['kernel'],

                                              kernel_size=conv_layers_hp[0]['ksize'],

                                              strides=conv_layers_hp[0]['stride'],

                                              padding=conv_layers_hp[0]['padding'],

                                              activation=tf.nn.relu, name="conv_1")

              pool_layers['1'] = tf.nn.max_pool(conv_layers['1'],

                                                ksize= [1] + pool_layers_hp[0]['ksize'] + [1],

                                                strides= [1] + pool_layers_hp[0]['stride'] + [1],

                                                padding= pool_layers_hp[0]['padding'], name='pool_1')

            else:

              conv_layers[str(i+1)] = tf.layers.conv2d(pool_layers[str(i)],  

                                                   filters=conv_layers_hp[i]['kernel'],                                       

                                                   kernel_size=conv_layers_hp[i]['ksize'],

                                                   strides=conv_layers_hp[i]['stride'],

                                                   padding=conv_layers_hp[i]['padding'],

                                                   activation=tf.nn.relu, name="conv_{}".format(i+1))

              pool_layers[str(i+1)] = tf.nn.max_pool(conv_layers[str(i+1)], ksize= [1] + pool_layers_hp[i]['ksize'] + [1],

                                               strides= [1] + pool_layers_hp[i]['stride'] + [1],  

                                               padding= pool_layers_hp[i]['padding'], name='pool_{}'.format(i+1))

 

 

                # convert multi-channel to one channel by flattening

          if average_pool: # if average-pooling is used, no extra fully-connected layer is created

            ave_pool_layer = tf.reduce_mean(pool_layers[str(n_blocks)], axis=[1,2])  # average the whole channel for every channel

            feature_dim = ave_pool_layer.shape[1]  # flatten the multi-channel to 1 channel

            ave_pool_flatten = tf.reshape(ave_pool_layer, [-1, feature_dim]) # fully connected layer

            fc = tf.layers.dense(ave_pool_flatten, fc_layers_hp, activation=tf.nn.relu, name="fc")  # fully connected layer

          else:  # if average-pooling is not used, a fully-connected layer is created after last pool layer and before last layer

            feature_dim = pool_layers[str(n_blocks)].shape[1] * pool_layers[str(n_blocks)].shape[2] * pool_layers[str(n_blocks)].shape[3] 

            pool_flat = tf.reshape(pool_layers[str(n_blocks)], [-1, feature_dim])  # flatten the multi-channels to 1 channel

            pool_flat_drop = tf.layers.dropout(pool_flat, pool_dropout_rate, training=training)

            fc = tf.layers.dense(pool_flat_drop, fc_layers_hp, activation=tf.nn.relu, name="fc")  # fully connected layer

 

          fc_drop = tf.layers.dropout(fc, fc_dropout_rate, training=training) 

        

      return fc_drop

       

        

  g = tf.Graph()

 

  with g.as_default():

    X = tf.placeholder(tf.float32, shape=(None, n_inputs), name="X")

    X_reshaped = tf.reshape(X, shape=[-1, height, width, channels])

    y = tf.placeholder(tf.float32, shape=(None, n_outputs), name="y")

    training = tf.placeholder_with_default(False, shape=(), name='training')

    alpha = tf.Variable([1., 1., 1., 1., 1.])

 

    # check if we have the same no of conv_layers and pool_layers

    assert len(conv_layers_hp) == len(pool_layers_hp), 'conv_layers and pool_layers should have the same length'

 

    # no of blocks of conv_layer + pool_layer

    n_blocks = len(conv_layers_hp)

 

    with tf.name_scope("dnn"):

      hidden_output_1 = build_cnn_layers(X_reshaped, n_blocks=n_blocks, var_scope_name='hidden_block_1')

   

      inputs_2 = tf.where(X_reshaped > 1 - 1e-2, X_reshaped, tf.zeros_like(X_reshaped))

      hidden_output_2 = build_cnn_layers(inputs=inputs_2, n_blocks=n_blocks, var_scope_name='hidden_block_2')

   

      inputs_3 = tf.where(X_reshaped > 1 - 1e-3, X_reshaped, tf.zeros_like(X_reshaped))

      hidden_output_3 = build_cnn_layers(inputs=inputs_3, n_blocks=n_blocks, var_scope_name='hidden_block_3')

   

      inputs_4 = tf.where(X_reshaped > 1 - 1e-4, X_reshaped, tf.zeros_like(X_reshaped))

      hidden_output_4 = build_cnn_layers(inputs=inputs_4, n_blocks=n_blocks, var_scope_name='hidden_block_4')

   

      inputs_5 = tf.where(X_reshaped > 1 - 1e-5, X_reshaped, tf.zeros_like(X_reshaped))

      hidden_output_5 = build_cnn_layers(inputs=inputs_5, n_blocks=n_blocks, var_scope_name='hidden_block_5')

       

        

#       hidden_outputs_stack = tf.stack([hidden_output_1, hidden_output_2, hidden_output_3,

#                                        hidden_output_4, hidden_output_5])   

#       hidden_output = tf.reshape(tf.matmul(tf.reshape(alpha, (1, 5)), tf.reshape(hidden_outputs_stack, (5, -1))),

#                                  (-1, 1000))

     

      hidden_output = alpha[0]*hidden_output_1 + alpha[1]*hidden_output_2 + alpha[2]*hidden_output_3 + \
                        alpha[3]*hidden_output_4 + alpha[4]*hidden_output_5

     

      logits = tf.layers.dense(hidden_output, n_outputs, activation=tf.nn.relu,

                               kernel_initializer=tf.contrib.layers.xavier_initializer(),

                               name="last") 

 

    with tf.name_scope("loss"):

      xentropy = tf.nn.sigmoid_cross_entropy_with_logits(labels=y,logits=logits)

      loss = tf.reduce_mean(xentropy)

       

      y_proba = tf.nn.sigmoid(logits)  # prediction

      correct = tf.equal(tf.round(y_proba), y)

      accuracy = tf.reduce_mean(tf.cast(correct, tf.float32))

 

    with tf.name_scope("train"):

      optimizer = tf.train.AdamOptimizer(learning_rate=learning_rate)

      training_op = optimizer.minimize(loss)

 

  return g, loss, accuracy, training_op, X, y, y_proba
